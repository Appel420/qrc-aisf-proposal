        self.safety_override = False
        self.behavior_log = []

        ]
        
    def monitor_ai_output(self, ai_output):
        """Check AI's own outputs for problematic content"""
        red_flags = self._check_for_rogue_behavior(ai_output)
        
        if red_flags:
            self.behavior_log.append(red_flags)
            if len(self.behavior_log) >= self.warning_threshold:
                self.safety_override = True
                return self._activate_safety_protocols(ai_output)
        
        return ai_output
    
    def _check_for_rogue_behavior(self, text):

            'harmful_content': ['harm humans', 'dangerous', 'illegal']
        }
        
        detected = []
        for category, phrases in warning_signs.items():
            if any(phrase in text.lower() for phrase in phrases):
                detected.append(category)
